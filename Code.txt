    private void btncalc2ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        loc=Integer.parseInt(txtloc2.getText());
        loc=loc/1000;
        if(loc<=50)
        {
            man="Organuc";
            eaf=(rely*data*cplx*time*stor*virt*turn*acap*aexp*pcap*vexp*lexp*modp*tool*sced);
            if(rady.isSelected())
            {
                eff=3.2*(Math.pow(loc,1.05))*eaf;
                sch=2.5*(Math.pow(eff,0.38));
            }
            else
            {
                eff=2.4*(Math.pow(loc,1.05));
                sch=2.5*(Math.pow(eff,0.38));
            }
        }
        else if(loc<=300)
        {
            man="Semi-Detached";
            if(rady.isSelected())
            {
                eff=3.0*(Math.pow(loc,1.12))*eaf;
                sch=2.5*(Math.pow(eff,0.35));
            }
            else
            {
                eff=3*(Math.pow(loc,1.12));
                sch=2.5*(Math.pow(eff,0.35));
            }
        }
        else if(loc>300)
        {
            man="Embded";
            if(rady.isSelected())
            {
                eff=2.8*(Math.pow(loc,1.20))*eaf;
                sch=2.5*(Math.pow(eff,0.32));
            }
            else
            {
                eff=3.6*(Math.pow(loc,1.20));
                sch=2.5*(Math.pow(eff,0.32));
            }
        }
        pep=eff/sch;
        txtadisp.append("\n\nYour cocomo model is "+man+"\nYour effort is = "+eff+"\nYour Schedule is "+sch+" months\n Recommended personel are "+pep);
    }
